local function createIntro()
    local introContainer = Instance.new("Frame", ScreenGui)
    introContainer.Size = UDim2.new(1, 0, 1, 0)
    introContainer.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    introContainer.BackgroundTransparency = 0
    introContainer.ZIndex = 100
    
    local intro = Instance.new("TextLabel", introContainer)
    intro.Size = UDim2.new(0.6, 0, 0.15, 0)
    intro.Position = UDim2.new(0.2, 0, 0.4, 0)
    intro.BackgroundTransparency = 1
    intro.Text = "Roblox Simple V2.0"
    intro.Font = Enum.Font.GothamBlack
    intro.TextColor3 = Color3.fromRGB(255, 165, 0)
    intro.TextScaled = true
    intro.ZIndex = 101
    
    local subText = Instance.new("TextLabel", introContainer)
    subText.Size = UDim2.new(0.4, 0, 0.1, 0)
    subText.Position = UDim2.new(0.3, 0, 0.55, 0)
    subText.BackgroundTransparency = 1
    subText.Text = "Enhanced Version"
    subText.Font = Enum.Font.GothamBold
    subText.TextColor3 = Color3.fromRGB(255, 255, 255)
    subText.TextScaled = true
    subText.ZIndex = 101
    subText.TextTransparency = 1
    
    -- Animate intro
    TweenService:Create(intro, TweenInfo.new(1.5, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {
        TextTransparency = 0,
        Position = UDim2.new(0.2, 0, 0.35, 0)
    }):Play()
    
    task.wait(1)
    
    TweenService:Create(subText, TweenInfo.new(1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
        TextTransparency = 0
    }):Play()
    
    task.wait(2)
    
    -- Fade out intro
    local fadeOut = TweenService:Create(introContainer, TweenInfo.new(1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
        BackgroundTransparency = 1
    })
    
    TweenService:Create(intro, TweenInfo.new(1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
        TextTransparency = 1
    }):Play()
    
    TweenService:Create(subText, TweenInfo.new(1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
        TextTransparency = 1
    }):Play()
    
    fadeOut:Play()
    fadeOut.Completed:Wait()
    introContainer:Destroy()
end

-- Function to create scrolling frame
local function createScrollingFrame(parent)
    local scrollFrame = Instance.new("ScrollingFrame", parent)
    scrollFrame.Size = UDim2.new(1, 0, 1, 0)
    scrollFrame.BackgroundTransparency = 1
    scrollFrame.BorderSizePixel = 0
    scrollFrame.ScrollBarThickness = 6
    scrollFrame.ScrollBarImageColor3 = Color3.fromRGB(255, 140, 0)
    
    local uiListLayout = Instance.new("UIListLayout", scrollFrame)
    uiListLayout.Padding = UDim.new(0, 10)
    uiListLayout.SortOrder = Enum.SortOrder.LayoutOrder
    
    return scrollFrame
end

-- Create tabs and pages
for i, tabName in ipairs(tabNames) do
    local tab = Instance.new("TextButton", tabBar)
    tab.Size = UDim2.new(1/#tabNames, 0, 1, 0)
    tab.Position = UDim2.new((i-1)/#tabNames, 0, 0, 0)
    tab.Text = tabName
    tab.Font = Enum.Font.GothamSemibold
    tab.TextColor3 = Color3.new(1, 1, 1)
    tab.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
    tab.TextScaled = true
    tab.BorderSizePixel = 0

    local page = Instance.new("Frame", frame)
    page.Size = UDim2.new(1, 0, 1, -85)
    page.Position = UDim2.new(0, 0, 0, 85)
    page.BackgroundColor3 = Color3.fromRGB(35, 35, 35)
    page.Visible = (i == 1)
    page.BorderSizePixel = 0
    
    local pageCorner = Instance.new("UICorner", page)
    pageCorner.CornerRadius = UDim.new(0, 0, 0, 12)

    local scrollFrame = createScrollingFrame(page)
    pages[tabName] = scrollFrame

    -- Enhanced tab hover effects
    tab.MouseEnter:Connect(function()
        if not page.Visible then
            TweenService:Create(tab, TweenInfo.new(0.2), {
                BackgroundColor3 = Color3.fromRGB(70, 70, 70)
            }):Play()
        end
    end)
    
    tab.MouseLeave:Connect(function()
        if not page.Visible then
            TweenService:Create(tab, TweenInfo.new(0.2), {
                BackgroundColor3 = Color3.fromRGB(50, 50, 50)
            }):Play()
        end
    end)
    
    tab.MouseButton1Click:Connect(function()
        for _, p in pairs(pages) do 
            p.Parent.Visible = false 
        end
        page.Visible = true
        
        -- Update tab colors
        for _, t in ipairs(tabBar:GetChildren()) do
            if t:IsA("TextButton") then
                TweenService:Create(t, TweenInfo.new(0.2), {
                    BackgroundColor3 = Color3.fromRGB(50, 50, 50)
                }):Play()
            end
        end
        
        TweenService:Create(tab, TweenInfo.new(0.2), {
            BackgroundColor3 = Color3.fromRGB(255, 105, 0)
        }):Play()
    end)
    
    -- Set first tab as active
    if i == 1 then
        tab.BackgroundColor3 = Color3.fromRGB(255, 105, 0)
    end
end

-- Enhanced button creation function
local function createEnhancedButton(parent, name, description, scriptCode, isExternal, hasInput)
    local buttonContainer = Instance.new("Frame", parent)
    buttonContainer.Size = UDim2.new(0.95, 0, 0, 70)
    buttonContainer.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
    buttonContainer.BorderSizePixel = 0
    
    local buttonCorner = Instance.new("UICorner", buttonContainer)
    buttonCorner.CornerRadius = UDim.new(0, 8)
    
    local btn = Instance.new("TextButton", buttonContainer)
    btn.Size = UDim2.new(1, 0, 0.6, 0)
    btn.Position = UDim2.new(0, 0, 0, 0)
    btn.Text = name
    btn.Font = Enum.Font.GothamBold
    btn.TextColor3 = Color3.new(1, 1, 1)
    btn.BackgroundTransparency = 1
    btn.TextScaled = true
    btn.TextXAlignment = Enum.TextXAlignment.Left
    
    local desc = Instance.new("TextLabel", buttonContainer)
    desc.Size = UDim2.new(1, -10, 0.4, 0)
    desc.Position = UDim2.new(0, 10, 0.6, 0)
    desc.Text = description or "Click to execute"
    desc.Font = Enum.Font.Gotham
    desc.TextColor3 = Color3.fromRGB(200, 200, 200)
    desc.BackgroundTransparency = 1
    desc.TextScaled = true
    desc.TextXAlignment = Enum.TextXAlignment.Left
    
    local input
    if hasInput then
        input = Instance.new("TextBox", buttonContainer)
        input.Size = UDim2.new(0.3, 0, 0.6, -10)
        input.Position = UDim2.new(0.7, 0, 0.2, 0)
        input.PlaceholderText = "Enter value"
        input.Text = ""
        input.BackgroundColor3 = Color3.fromRGB(70, 70, 70)
        input.TextColor3 = Color3.new(1, 1, 1)
        input.Font = Enum.Font.Gotham
        input.TextScaled = true
        input.BorderSizePixel = 0
        
        local inputCorner = Instance.new("UICorner", input)
        inputCorner.CornerRadius = UDim.new(0, 4)
    end
    
    -- Button hover effects
    btn.MouseEnter:Connect(function()
        TweenService:Create(buttonContainer, TweenInfo.new(0.2), {
            BackgroundColor3 = Color3.fromRGB(65, 65, 65)
        }):Play()
    end)
    
    btn.MouseLeave:Connect(function()
        TweenService:Create(buttonContainer, TweenInfo.new(0.2), {
            BackgroundColor3 = Color3.fromRGB(50, 50, 50)
        }):Play()
    end)
    
    btn.MouseButton1Click:Connect(function()
        TweenService:Create(buttonContainer, TweenInfo.new(0.1), {
            BackgroundColor3 = Color3.fromRGB(255, 105, 0)
        }):Play()
        
        task.wait(0.1)
        
        TweenService:Create(buttonContainer, TweenInfo.new(0.1), {
            BackgroundColor3 = Color3.fromRGB(65, 65, 65)
        }):Play()
        
        -- Execute script
        local success, errorMsg = pcall(function()
            if hasInput and input then
                local val = tonumber(input.Text)
                if val then
                    loadstring(scriptCode:gsub("{{val}}", tostring(val)))()
                else
                    warn("Invalid input value")
                end
            else
                if isExternal then
                    loadstring(game:HttpGet(scriptCode))()
                else
                    loadstring(scriptCode)()
                end
            end
        end)
        
        if not success then
            warn("Script execution failed: " .. tostring(errorMsg))
        end
    end)
    
    return buttonContainer
end

    if key.KeyCode == Enum.KeyCode.RightShift then
        guiVisible = not guiVisible
        
        if guiVisible and not frame.Visible then
            frame.Visible = true
            frame.Size = UDim2.new(0, 0, 0, 0)
            frame.Position = UDim2.new(0.5, 0, 0.5, 0)
            
            TweenService:Create(frame, TweenInfo.new(0.5, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {
                Size = UDim2.new(0, 600, 0, 450),
                Position = UDim2.new(0.5, -300, 0.5, -225)
            }):Play()
        elseif not guiVisible and frame.Visible then
            TweenService:Create(frame, TweenInfo.new(0.3, Enum.EasingStyle.Back, Enum.EasingDirection.In), {
                Size = UDim2.new(0, 0, 0, 0),
                Position = UDim2.new(0.5, 0, 0.5, 0)

end)
