-- REAL MONEY INJECTION SYSTEM
local function injectRealGameMoney()
    statusLabel.Text = "üîÑ INJECTING REAL GAME MONEY..."
    statusLabel.TextColor3 = Color3.fromRGB(255, 255, 0)
    
    wait(1)
    
    local success = false
    
    -- METHOD 1: Find and modify ACTUAL game currency
    pcall(function()
        -- Look for the REAL money system used for purchases
        local moneyNames = {"Cash", "Money", "Coins", "Currency", "Gold", "Points"}
        
        for _, moneyName in pairs(moneyNames) do
            -- Check player folder
            local playerMoney = player:FindFirstChild(moneyName)
            if playerMoney and (playerMoney:IsA("IntValue") or playerMoney:IsA("NumberValue")) then
                playerMoney.Value = 999999999
                success = true
                statusLabel.Text = "‚úÖ REAL MONEY ADDED: " .. moneyName .. "\nAmount: $999,999,999"
                break
            end
            
            -- Check leaderstats
            local leaderstats = player:FindFirstChild("leaderstats")
            if leaderstats then
                local statsMoney = leaderstats:FindFirstChild(moneyName)
                if statsMoney and (statsMoney:IsA("IntValue") or statsMoney:IsA("NumberValue")) then
                    statsMoney.Value = 999999999
                    success = true
                    statusLabel.Text = "‚úÖ REAL MONEY ADDED: " .. moneyName .. "\nAmount: $999,999,999"
                    break
                end
            end
        end
    end)
    
    -- METHOD 2: Create purchase currency if not found
    if not success then
        pcall(function()
            local purchaseCurrency = Instance.new("IntValue")
            purchaseCurrency.Name = "Cash"
            purchaseCurrency.Value = 999999999
            purchaseCurrency.Parent = player
            
            success = true
            statusLabel.Text = "‚úÖ PURCHASE CURRENCY CREATED\nCash: $999,999,999"
        end)
    end
    
    if success then
        statusLabel.TextColor3 = Color3.fromRGB(0, 255, 0)
    else
        statusLabel.Text = "‚ùå FAILED: Could not inject money\nTry again or use different method"
        statusLabel.TextColor3 = Color3.fromRGB(255, 0, 0)
    end
    
    return success
end
